all groups with less or equals student count:

SELECT groups.* FROM
	(SELECT group_id, COUNT(group_id) AS students_count
	FROM public.students
	GROUP BY group_id) as foo
LEFT JOIN public.groups
ON group_id = groups.id
WHERE students_count <= 25;


all students related to the course with a given name:

SELECT students.* FROM public.students_courses_relation
LEFT JOIN public.students
ON students.id = student_id
WHERE course_id =
(SELECT id FROM public.courses
WHERE name = 'math');
